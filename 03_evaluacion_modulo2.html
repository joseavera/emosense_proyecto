¿Qué hacemos en este módulo?

En el Módulo 1 entrenaste un clasificador de emociones con una configuración fija. Ahora compararemos 6 combinaciones distintas de hiperparámetros para encontrar cuál produce el mejor F1-Score en EmoSense.

La clave de este módulo es el cross-validation estratificado: en lugar de evaluar con una sola partición 80/20, el modelo se entrena y prueba 5 veces con distintos datos. Esto da una estimación mucho más confiable del rendimiento real.

También usaremos Pipeline de sklearn para encadenar TF-IDF + Random Forest en un objeto único. Esto previene la contaminación de datos: el vocabulario se aprende dentro de cada pliegue, no antes.

📐 El pipeline del módulo 2
Entrada Texto crudo en español
→
Paso 1 limpiar_texto()
→
Paso 2 — hiperparámetro TfidfVectorizer
→
Paso 3 — hiperparámetro RandomForestClassifier
→
Salida Emoción predicha

Pipeline() envuelve los pasos 2 y 3. cross_val_score() repite los pasos 1→2→3 cinco veces con distintas particiones para evaluar qué tan bien generaliza cada configuración.

def limpiar_texto(texto):
    texto = texto.lower()
    texto = re.sub(r"[^a-záéíóúüñ\s]", " ", texto)
    texto = re.sub(r"\s+", " ", texto).strip()
    return texto

Bloque 3 · Función de comparación con Pipeline + CV
def comparar_configuraciones(textos, etiquetas):
    """
    Entrena y evalúa cada config con
    cross-validation estratificada de 5 pliegues.
    Usa Pipeline para evitar contaminación de datos.
    """
    textos_limpios = [
        limpiar_texto(t) for t in textos
    ]

    cv = StratifiedKFold(
        n_splits     = N_SPLITS,
        shuffle      = True,
        random_state = RANDOM_STATE,
    )

    resultados = []
    sep = "=" * 64

    print(sep)
    print("  EMOSENSE — COMPARACIÓN DE CONFIGURACIONES")
    print(f"  Cross-Validation: {N_SPLITS} pliegues")
    print(sep)

    for cfg in CONFIGURACIONES:

        # Pipeline encadena TF-IDF + Random Forest
        # El vocabulario se ajusta DENTRO de cada pliegue
        pipeline = Pipeline([
            ("tfidf", TfidfVectorizer(
                max_features = cfg["max_features"],
                ngram_range  = cfg["ngram_range"],
            )),
            ("clf", RandomForestClassifier(
                n_estimators     = cfg["n_estimators"],
                max_depth        = cfg["max_depth"],
                min_samples_leaf = cfg["min_leaf"],
                random_state     = RANDOM_STATE,
            )),
        ])

        scores = cross_val_score(
            pipeline,
            textos_limpios,
            etiquetas,
            cv      = cv,
            scoring = "f1_weighted",
        )

        media = scores.mean()
        std   = scores.std()

        resultados.append({
            "nombre"  : cfg["nombre"],
            "media"   : media,
            "std"     : std,
            "cfg"     : cfg,
        })

        print(
            f"  {cfg['nombre']:<28} "
            f"F1 = {media:.4f} ± {std:.4f}"
        )

    print(sep)
    mejor = max(resultados, key=lambda r: r["media"])
    print(f"\n  🏆 MEJOR CONFIG : {mejor['nombre']}")
    print(f"     F1 CV media  : {mejor['media']:.4f}")
    print(f"     F1 CV std    : {mejor['std']:.4f}")

    return resultados, mejor
Bloque 4 · Guardar modelo óptimo + gráfica + main
def guardar_mejor(mejor, textos, etiquetas):
    """
    Reentrena la mejor config con TODOS los datos
    (sin partición train/test) y guarda el pipeline.
    Este archivo .pkl lo usará el Módulo 3.
    """
    textos_limpios = [limpiar_texto(t) for t in textos]
    cfg = mejor["cfg"]

    pipeline_final = Pipeline([         # ← con TODOS los datos
        ("tfidf", TfidfVectorizer(
            max_features = cfg["max_features"],
            ngram_range  = cfg["ngram_range"],
        )),
        ("clf", RandomForestClassifier(
            n_estimators     = cfg["n_estimators"],
            max_depth        = cfg["max_depth"],
            min_samples_leaf = cfg["min_leaf"],
            random_state     = RANDOM_STATE,
        )),
    ])
    pipeline_final.fit(textos_limpios, etiquetas)

    if not os.path.exists("modelos"):
        os.makedirs("modelos")

    ruta = os.path.join("modelos", "modelo_optimo.pkl")
    joblib.dump(pipeline_final, ruta)
    print(f"\n  ✅ Modelo guardado → {ruta}")
    print("  El Módulo 3 lo cargará con joblib.load()")


def graficar_comparacion(resultados, carpeta="graficas"):
    """Gráfica de barras de F1 CV por configuración."""
    if not os.path.exists(carpeta):
        os.makedirs(carpeta)

    nombres = [r["nombre"] for r in resultados]
    medias  = [r["media"]  for r in resultados]
    stds    = [r["std"]    for r in resultados]

    mejor_idx = medias.index(max(medias))
    colores   = [
        "#7c6af7" if i == mejor_idx else "#3a3a48"
        for i in range(len(nombres))
    ]

    fig, ax = plt.subplots(figsize=(10, 5))
    ax.bar(
        range(len(nombres)), medias,
        color=colores, yerr=stds,
        capsize=5, alpha=0.88, edgecolor="#1e1e24",
    )

    for i, (m, s) in enumerate(zip(medias, stds)):
        ax.text(i, m + s + 0.01,
                f"{m:.3f}", ha="center",
                fontsize=9, fontweight="bold")

    ax.set_xticks(range(len(nombres)))
    ax.set_xticklabels(
        nombres, rotation=20,
        ha="right", fontsize=9,
    )
    ax.set_ylim(0.5, 1.05)
    ax.set_ylabel("F1-Score weighted (CV)")
    ax.set_title(
        "EmoSense — F1 por configuración\n"
        "(media ± desv. estándar · 5 pliegues)",
        fontsize=11,
    )
    ax.axhline(0.80, color="gray",
               linestyle="--", linewidth=0.8)
    ax.grid(axis="y", alpha=0.3)
    plt.tight_layout()

    ruta = os.path.join(carpeta, "comparacion_configs.png")
    plt.savefig(ruta, dpi=120, bbox_inches="tight")
    print(f"  Gráfica guardada → {ruta}")
    plt.show()


def main():
    print("\n" + "=" * 64)
    print("  EMOSENSE — MÓDULO 2: OPTIMIZACIÓN")
    print("=" * 64 + "\n")

    textos, etiquetas = crear_dataset()   # del Módulo 1

    print("    modelos/modelo_optimo.pkl")
